pipeline {
    agent any
    
    environment {
        APP_NAME = "devops-internship-app"
        VERSION = "1.0.${BUILD_NUMBER}"
        DOCKER_IMAGE = "your-docker-username/${APP_NAME}:${VERSION}"
    }
    
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', 
                    url: 'https://github.com/your-username/your-repo.git'
            }
        }
        
        stage('Install Dependencies') {
            steps {
                sh 'npm install'
            }
        }
        
        stage('Test') {
            steps {
                sh 'npm test'
            }
            
            post {
                always {
                    junit '**/test-results.xml' 
                }
            }
        }
        
        stage('Build Docker Image') {
            steps {
                script {
                    sh "docker build -t ${DOCKER_IMAGE} ."
                }
            }
        }
        
        stage('Deploy to Test') {
            steps {
                script {
                    // For local testing with Docker
                    sh "docker run -d -p 3000:3000 --name ${APP_NAME}-test ${DOCKER_IMAGE}"
                    
                    // Or for Kubernetes:
                    // sh "kubectl apply -f k8s/test-deployment.yaml"
                }
            }
        }
    }
    
    post {
        always {
            cleanWs()
        }
        success {
            echo 'Pipeline completed successfully!'
        }
        failure {
            echo 'Pipeline failed!'
        }
    }
}
